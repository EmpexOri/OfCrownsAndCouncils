namespace = sax_monarchy

sax_monarchy.1 = {
	type = country_event
	placement = root

	title = sax_monarchy.1.t
	desc = sax_monarchy.1.d
	flavor = sax_monarchy.1.f

	gui_window = event_window_1char_tabloid
	icon = "gfx/interface/icons/law_icons/monarchy.dds"

    left_icon = scope:new_king

	duration = 3
    cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		# Fired via on_action only
	}

	immediate = {
		ruler = { save_scope_as = new_king }
		root = { save_scope_as = hannover_scope }
	}

	option = {
		name = sax_monarchy.1.a
		default_option = yes
		hidden_effect = {
			if = {
				limit = {
					game_date >= 1833.12.26
				}
				create_character = {
					template = SAX_Luidolf_IV
				}
			}
		}

		custom_tooltip = {
			text = sax_dynastic_succession_tt
		}

	}
}

namespace = sax_luddites

sax_luddites.1 = {
	type = country_event
	placement = root

	title = sax_luddites.1.t
	desc = sax_luddites.1.d
	flavor = sax_luddites.1.f

	gui_window = event_window_1char_tabloid
	icon = "gfx/interface/icons/law_icons/monarchy.dds"

    left_icon = scope:sax_rural_leader

	duration = 3
    cooldown = { days = very_long_modifier_time }

	trigger = {
		# Fired via on_action only
	}

	immediate = {
		ruler = { save_scope_as = sax_ruler }
		root = { save_scope_as = hannover_scope }
        ig:ig_rural_folk = {
            leader = {
                save_scope_as = sax_rural_leader
            }
        }
	}

	option = {
		name = sax_luddites.1.a
		default_option = yes
		scope:luddite_movement_scope = {
			add_modifier = {
				name = modifier_luddite_unrest
			}
		}
	}
}
