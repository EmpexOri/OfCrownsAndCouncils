	}

    title = polish_queen_rise.3.t
    desc = polish_queen_rise.3.d
    flavor = polish_queen_rise.3.f
    icon = "gfx/interface/icons/event_icons/event_protest.dds"

    trigger = { }

    option = {
        name = polish_queen_rise.3.a
        default_option = yes

        add_modifier = {
            name = stability_in_revolution
            years = 20
            is_decaying = yes
        }

        activate_law = law_type:law_parliamentary_republic

        if = {
            limit = {
                OR = {
                    has_law_or_variant = law_type:law_autocracy
                    has_law_or_variant = law_type:law_oligarchy
                }
            }
            activate_law = law_type:law_wealth_voting
        }
    }
    option = {
        name = polish_queen_rise.3.b
        add_modifier = {
            name = stability_in_military_revolution
            years = 20
            is_decaying = yes
        }
        activate_law = law_type:law_presidential_republic
        activate_law = law_type:law_autocracy
    }
