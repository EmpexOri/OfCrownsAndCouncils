	should_be_pinned_by_default = yes

je_empire_of_the_horn = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_ethiopia

	is_shown_in_lobby = {
		c:ETH = THIS
	}

    is_shown_when_inactive = {
		OR = {
            c:ETH ?= THIS
        }
    }

    possible = {
        OR = {
            c:ETH ?= THIS
        }
		AND = {
			has_modifier = modifier_unified_negus
			has_modifier = modifier_unified_government_eth
			has_technology_researched = pharmaceuticals
			has_technology_researched = colonization
		}
    }

	modifiers_while_active = {
		#modifier_industrialing_eth
	}

	immediate = {
		add_technology_researched = quinine
		trigger_event = {
			id = ethiopian_empire.1
		}
	}

	complete = {
		OR = {
			country_or_subject_owns_entire_state_region = STATE_KENYA
			country_or_subject_owns_entire_state_region = STATE_RIFT_VALLEY
			country_or_subject_owns_entire_state_region = STATE_UGANDA
		}
	}

	on_complete = {
		add_modifier = {
			name = modifier_the_new_empire
			years = 10
		}
		show_as_tooltip = {
			set_country_type = recognized
		}
	}

	weight = 10000
	should_be_pinned_by_default = yes
}
