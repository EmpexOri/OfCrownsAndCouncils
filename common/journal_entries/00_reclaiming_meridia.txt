je_reclaiming_meridia = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	group = je_french_america

	is_shown_in_lobby = {
		c:W38 = THIS
    }

    scripted_button = je_sardo_reclaim_meridia_button

	complete = {
		OR = {
            c:BRZ = { is_subject_of = ROOT }
            c:BRZ = { 
                gdp_ownership_ratio = {
                    target = ROOT
                    value > 0.2
                }
            }
        }
	}

	on_complete = {
		set_variable = reclaiming_meridia_completed
        trigger_event = {
            id = reclaiming_meridia.2
            popup = yes
        }
        custom_tooltip = {
			text = company_possible_tt
		}
	}

    weight = 10000
}
